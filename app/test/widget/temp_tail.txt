    await tester.pumpWidget(
      UncontrolledProviderScope(
        container: container,
        child: const MaterialApp(home: SettingsScreen()),
      ),
    );

    await tester.tap(find.byType(Switch));
    await tester.pumpAndSettle();

    final items = await repo.getAllItems();
    expect(items, isEmpty);
    expect(container.read(demoModeProvider), isFalse);

    final prefs = await SharedPreferences.getInstance();
    expect(prefs.getBool('demo_mode_enabled'), isFalse);
  });

  testWidgets('First manual add while demo on flips demo off, persists flags, invalidates items', (tester) async {
    SharedPreferences.setMockInitialValues({});
    final repo = InMemoryItemRepository();
    await repo.init();
    final container = ProviderContainer(overrides: [
      itemRepositoryProvider.overrideWithValue(repo),
      demoModeProvider.overrideWith((ref) => true),
      hasManualItemsProvider.overrideWith((ref) => false),
    ]);
    addTearDown(container.dispose);

    int invalidations = 0;
    final sub = container.listen(itemsFutureProvider, (previous, next) {
      invalidations++;
    });
    addTearDown(sub.close);

    await tester.pumpWidget(
      UncontrolledProviderScope(
        container: container,
        child: const MaterialApp(home: ItemFormScreen()),
      ),
    );

    await tester.pumpAndSettle();
    expect(container.read(demoModeProvider), isTrue);
    expect(find.widgetWithText(ElevatedButton, 'Add Item'), findsOneWidget);

    final nameField = find.descendant(
      of: find.ancestor(of: find.text('Name *'), matching: find.byType(Column)),
      matching: find.byType(TextFormField),
    );
    await tester.enterText(nameField, 'Apples');
    final addButton = find.widgetWithText(ElevatedButton, 'Add Item');
    await tester.ensureVisible(addButton);
    await tester.tap(addButton);
    await tester.pumpAndSettle();

    // Allow any pending microtasks from post-save callbacks to flush
    await tester.pump(const Duration(milliseconds: 10));

    final items = await repo.getAllItems();
    expect(items.length, 1);

    expect(container.read(demoModeProvider), isFalse);
    expect(container.read(hasManualItemsProvider), isTrue);

    final prefs = await SharedPreferences.getInstance();
    expect(prefs.getBool('demo_mode_enabled'), isFalse);
    expect(prefs.getBool('has_manual_items'), isTrue);

    expect(invalidations, greaterThanOrEqualTo(1));
  });

  testWidgets('Adding item after demo off updates manual flag without provider mutation errors', (tester) async {
    final repo = InMemoryItemRepository();
    final container = ProviderContainer(overrides: [

  });
